pipeline {
    agent any

    tools {
        jdk 'JAVA_HOME'
        ant 'ANT'
    }

    stages {

        stage('Checkout Code') {
            steps {
                git branch: 'master',
                    credentialsId: 'github-creds',
                    url: 'https://github.com/NamarataSharma/V3Practice.git'
            }
        }

        stage('Run Provar Tests') {
            steps {
                bat 'ant -f ANT/build.xml'
            }
        }
    }

    post {
        always {
            junit allowEmptyResults: true, testResults: 'ANT/Results/*.xml'
        }
    }
}
